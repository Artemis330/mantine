# use-document-title

<PageHeader
  source="mantine-hooks/src/hooks/use-document-title/use-document-title.ts"
  docs="hooks/use-document-title.mdx"
  package="@mantine/hooks"
/>

Sets `document.title` property with `React.useLayoutEffect` hook.

```tsx
import { useDocumentTitle } from '@mantine/hooks';
```

## Usage

```tsx
import React from 'react';
import { useDocumentTitle } from '@mantine/hooks';

function ApplicationRoute() {
  useDocumentTitle('My application route');
  return <div />;
}
```
